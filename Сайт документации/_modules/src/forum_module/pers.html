  def UpdatePers(self, new_health):
    """
    Обновление персонажа

    Args:
      new_health (int): Число которое хранит в себе знчаение здоровья которое отбавляется или востанавливается

    Returns:
      (str): Сообщение об успешном изменении здоровья персонажа
    """
    self.health = self.health + new_health
    return f"Успешное изменение здоровья персонажа. Текущее здоровье {self.health}"
